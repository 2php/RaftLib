1 /**
2  * CmdArgs - A very simple command line arguments src file, can be included
3  * in C++ and C files using the appropriate function calls. In order to
4  * use this to process your arguments simply do the following:
5  * //create a CmdArgs object
6  * CmdArgs *args = new CmdArgs(std::string(argv[0])); //sets program name for object
7  * //define static variables for options
8  * static bool myFirstOption = false; //assigning var gives default value
9  * //add options to CmdArgs
10  * cmd->addOption(new Option<bool>(&myFirstOption,"-mFO","An option"));
11  * //general form of
12  * cmd->addOption(new Option<type>(var mem location, text flag, text description));
13  * //give cmd line args to CmdArgs
14  * cmd->processArgs(argc,argv)
15  * .....now you can use the variables as you see fit
16  *
17  * @author Jonathan Beard
18  * @version Last modified 27 May 2012
19  * @contact jbeard@wustl.edu
20  */
