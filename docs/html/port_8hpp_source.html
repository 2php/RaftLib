75 /** continue recursion **/
76 template < class T,
77  class PORT,
78  class PORTNAME,
79  class... PORTNAMES >
80 struct port_helper< T, PORT, PORTNAME, PORTNAMES... >
81 {
82  static void add_port( PORT &port,
83  PORTNAME &&portname,
84  PORTNAMES&&... portnames )
85  {
86  port.template add_port< T >( portname );
87  port_helper< T, PORT, PORTNAMES... >::add_port( port,
88  std::forward< PORTNAMES >( portnames )... );
89  return;
90  }
91 };

155  /**
156  * addPorts - add ports for an existing buffer, basically
157  * allocate buffers in place. These also won't be able
158  * to be resized.
159  * @param n_ports - const std::size_t
160  */
161  template < class T >
162  bool addInPlacePorts( const std::size_t n_ports )
163  {
164  T *existing_buff_t( reinterpret_cast< T* >( alloc_ptr ) );
165  std::size_t length( alloc_ptr_length / sizeof( T ) );
166  const std::size_t inc( length / n_ports );
167  const std::size_t adder( length % n_ports );
168 
169  using index_type = std::remove_const_t<decltype(n_ports)>;
170  for( index_type index( 0 ); index < n_ports; index++ )
171  {
172  const std::size_t start_index( index * inc );
173  PortInfo pi( typeid( T ),
174  (void*)&( existing_buff_t[ start_index ] ) /** pointer **/,
175  inc + ( index == (n_ports - 1) ? adder : 0 ),
176  start_index );
177  pi.my_kernel = kernel;
178  const std::string name( std::to_string( index ) );
179  pi.my_name = name;
180  /** gotta initialize the maps to copy stuff to/from **/
181  (this)->initializeConstMap< T >( pi );
182  (this)->initializeSplit< T >( pi );
183  (this)->initializeJoin< T >( pi );
184  portmap.map.insert( std::make_pair( name, pi ) );
185  }
186  return( true );
187  }
