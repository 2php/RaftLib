62 kernel_container::container_run( kernel_container &container )
63 {
64  bool shutdown( false );
65  auto &input_buffer( container.getInputQueue() );
66  auto &output_buffer( container.getOutputQueue() );
67  while( ! shutdown || container.preempted_kernel_pool.size() > 0 )
68  {
69  if( container.getInputQueue().size() > 0 )
70  {
71  sched_cmd_t new_cmd;
72  input_buffer.pop< sched_cmd_t >( new_cmd );
73  switch( new_cmd.cmd )
74  {
75  case( schedule::add ):
76  {
77  assert( new_cmd.kernel != nullptr );
78  //FIXME: hacked this so it'll compile, need to fix preempt state
79  //const auto ret_val( setPreemptState( new_cmd.kernel ) );
80  const auto ret_val( 0 );
81  switch( ret_val )
82  {
83  case( 0 /* newly scheduled kernel */ ):
84  {
85  bool done( false );
86  auto &out_cmd( output_buffer.allocate< sched_cmd_t >() );
87  Schedule::kernelRun( new_cmd.kernel, done );
88  out_cmd.cmd = ( done ? schedule::kernelfinished :
89  schedule::reschedule );
90  out_cmd.kernel = new_cmd.kernel;
91  output_buffer.send();
92  /** clean-up buffer and recycle head of FIFO **/
93  }
94  break;
95  case( 1 /* kernel preempted */ ):
96  {
97  container.preempted_kernel_pool.push( new_cmd.kernel );
98  }
99  break;
100  default:
101  assert( false );
102  }
103  }
104  break;
105  case( schedule::shutdown ):
106  {
107  /** just in case, a sanity check here **/
108  shutdown = true;
109  }
110  break;
111  default:
112  {
113  std::cerr << "Invalid signal: " <<
114  schedule::sched_cmd_str[ new_cmd.cmd ] << "\n";
115  assert( false );
116  }
117  }
118  }
119  /** try these kernels again **/
120  if( container.preempted_kernel_pool.size() > 0 )
121  {
122  //auto * const kernel( container.preempted_kernel_pool.front() );
123  container.preempted_kernel_pool.pop();
124  //restore( kernel );
125  /** after this it'll longjmp to the running state **/
126  }
127  }
128 }
