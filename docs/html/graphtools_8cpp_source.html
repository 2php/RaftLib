94 void
95 GraphTools::__BFS( std::queue< raft::kernel* > &queue,
96  std::set< raft::kernel* > &visited_set,
97  edge_func func,
98  void *data,
99  bool connected_error )
100 {
101  while( queue.size() > 0 )
102  {
103  auto *k( queue.front() );
104  queue.pop();
105  if( k == nullptr ) break;
106  /** iterate over all out-edges **/
107  /** 1) get lock **/
108  while( ! k->output.portmap.mutex_map.try_lock() )
109  {
110  std::this_thread::yield();
111  }
112  //we have lock, continue
113  /** 2) get map **/
114  std::map< std::string, PortInfo > &map_of_ports( k->output.portmap.map );
115  for( auto &port : map_of_ports )
116  {
117  PortInfo &source( port.second );
118  /** get dst edge to call function on **/
119  if( source.other_kernel != nullptr )
120  {
121  PortInfo &dst(
122  source.other_kernel->input.getPortInfoFor( source.other_name ) );
123  func( source, dst, data );
124  }
125  else
126  if( connected_error )
127  {
128  std::stringstream ss;
129  ss << "Unconnected port detected at " <<
130  common::printClassName( *k ) <<
131  "[ \"" <<
132  source.my_name << " \"], please fix and recompile.";
133  k->output.portmap.mutex_map.unlock();
134  throw PortException( ss.str() );
135  }
136  /** if the dst kernel hasn't been visited, visit it **/
137  if( visited_set.find( source.other_kernel ) == visited_set.end() )
138  {
139  queue.push( source.other_kernel );
140  visited_set.insert( source.other_kernel );
141  }
142  }
143  k->output.portmap.mutex_map.unlock();
144  }
145  return;
146 }
