65  /**
66  * link - this comment goes for the next 4 types of link functions,
67  * which basically do the exact same thing. The template function
68  * takes a single param order::spec which is exactly as the name
69  * implies, the order of the queue linking the two kernels. The
70  * various functions are needed to specify different ordering types
71  * each of these will be commented separately below. This function
72  * assumes that Kernel 'a' has only a single output and raft::kernel 'b' has
73  * only a single input otherwise an exception will be thrown.
74  * @param a - raft::kernel*, src kernel
75  * @param b - raft::kernel*, dst kernel
76  * @throws AmbiguousPortAssignmentException - thrown if either src or
77  * dst have more than
78  * a single port to link.
79  * @return kernel_pair_t - references to src, dst kernels.
80  */
81  template < raft::order::spec t = raft::order::in >
82  kernel_pair_t link( raft::kernel *a,
83  raft::kernel *b,
84  const std::size_t buffer = 0 )
85  {
86  updateKernels( a, b );
87  PortInfo *port_info_a;
88  try{
89  port_info_a = &(a->output.getPortInfo());
90  }
91  catch( PortNotFoundException &ex )
92  {
93  portNotFound( true,
94  ex,
95  a );
96  }
97  port_info_a->fixed_buffer_size = buffer;
98  PortInfo *port_info_b;
99  try{
100  port_info_b = &(b->input.getPortInfo());
101  }
102  catch( PortNotFoundException &ex )
103  {
104  portNotFound( false,
105  ex,
106  b );
107  }
108  port_info_b->fixed_buffer_size = buffer;
109 
110  join( *a, port_info_a->my_name, *port_info_a,
111  *b, port_info_b->my_name, *port_info_b );
112  set_order< t >( *port_info_a, *port_info_b );
113  return( kernel_pair_t( a, b ) );
114  }

116  /**
117  * link - same as function above save for the following differences:
118  * kernel a is assumed to have multiple ports and the one we wish
119  * to link with raft::kernel b is a_port. raft::kernel b is assumed to have a

121  * @param a - raft::kernel *a, can have multiple ports
122  * @param a_port - port within raft::kernel a to link
