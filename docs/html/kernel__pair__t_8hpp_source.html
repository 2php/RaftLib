35 /**
36  * basic idea here is that we want a return type
37  * for the link() function calls as well as the
38  * operator += overload on the map that enables
39  * return of multiple compute kernels for the
40  * user to stitch together in another invocation
41  * to add more kernels to the map. What we also
42  * want to ensure is that there is no way for
43  * the user to delete one of these kernels which
44  * could create for some interesting bugs.
45  * the std::pair seems like an obvious choice
46  * to return, however creating a container as part
47  * of the std::pair is a bit hacky so we'll use the
48  * container directly, the next consideration is what
49  * type of container to use inside that std::pair
50  * which should be expandable. Again, the obvious
51  * choice seems like it should be the std::vector
52  * which needs to hold references so they can't
53  * be easily deleted..which means we'll have to
54  * use std::reference_wrapper as well.n
55  */
56 class kernel_pair_t
57 {
58  /** container with ref wrapper **/
59  using kernel_pair_t_container =
60  std::vector< std::reference_wrapper< raft::kernel> >;
61 
62 public:
63  /**
64  * define iterator type publicly
65  */
66  using kernel_iterator_type = kernel_pair_t_container::iterator;
67  /**
68  * endpoint ret type is a std::pair with
69  * two iterators, one for begin, the second
70  * for end
71  */
72  using endpoint_ret_type =
73  std::pair< kernel_iterator_type,
74  kernel_iterator_type >;
75  /**
76  * define a size type that matches the container
77  * type, whatever that container type may end
78  * up being
79  */
80  using size_type = typename kernel_pair_t_container::size_type;
81 
82  /**
83  * kernel_pair_t - default constructor, simply
84  * reserves some space for the container, assumes
85  * container has a reserve(xx) function to call
86  * in the first place...likely need to add a
87  * enable_if to make sure that is always the case
88  */
89  kernel_pair_t();
90 
91  /**
92  * kernel_pair_t - construct by first calling the
93  * base constructor, then insert into the containers
94  * with the parameters of this constructor.
95  * @param src - raft;:kernel, source kernel
96  * @param dst - raft::kernel, destination kernel
97  */
98  kernel_pair_t( raft::kernel * const src,
99  raft::kernel * const dst );
100 
101  /**
102  * kernel_pair_t - construct by first calling the
103  * base constructor, then insert into the containers
104  * with the parameters of this constructor.
105  * @param src - raft;:kernel, source kernel
106  * @param dst - raft::kernel, destination kernel
107  */
108  kernel_pair_t( raft::kernel &src,
109  raft::kernel &dst );
110 
111  /**
112  * getSrc - return a std::pair object with iterators
113  * to the source and destination of the list of
114  * sources added in the last map addition. Again,
115  * pair.first maps ot begin(), and pair.second maps
116  * to end();
117  * @return endpoint_ret_type
118  */
119  kernel_pair_t::endpoint_ret_type getSrc();
120  /**
121  * getSrcSize - returns the size of the source
122  * container. This is the number of kernels
123  * within the last map addition.
124  * @return size_type - number of kernels in source
125  * container
126  */
127  kernel_pair_t::size_type getSrcSize() noexcept;
128  /**
129  * getDst - return a std::pair object with iterators
130  * to the dst list of kernels added in the last map
131  * addition. Again, pair.first maps ot begin(), and
132  * pair.second maps to end();
133  * @return endpoint_ret_type
134  */
135  kernel_pair_t::endpoint_ret_type getDst();
136  /**
137  * getDstSize - returns the size of the destination
138  * container. This is the number of kernels
139  * within the last map addition.
140  * @return size_type - number of kernels in source
141  * container
142  */
143  kernel_pair_t::size_type getDstSize() noexcept;
144  /**
145  * addSrc - add a source kernel to this pair object
146  * which is retreivable by the getSrc. If this object
147  * disappears (is deallocated) after you add it then
148  * bad things might happen...don't delete them till
149  * you've disposed of these objects.
150  * @param k - raft::kernel&
151  */
152  void addSrc( raft::kernel &k ) noexcept;
153  /**
154  * addDst - add a destination kernel to this pair object
155  * which is retreivable by the getDst. If this object
156  * disappears (is deallocated) after you add it then
157  * bad things might happen...don't delete them till
158  * you've disposed of these objects.
159  * @param k - raft::kernel&
160  */
161  void addDst( raft::kernel &k ) noexcept;
162  /**
163  * clearSrc - does exactly what it says, clears out
164  * the list of kernels, does not however destoy them
165  * so they're still valid kernels, just not available
166  * to the list anymore
167  */
168  void clearSrc() noexcept;
169  /**
170  * clearDst - does exactly what it says, clears out
171  * the list of kernels, does not however destoy them
172  * so they're still valid kernels, just not available
173  * to the list anymore
174  */
175  void clearDst() noexcept;
176 
177 private:
178  /** type is determined by using type aliases above the first public: **/
179  kernel_pair_t_container source;
180  kernel_pair_t_container destination;
181 };
