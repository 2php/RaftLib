99 void
100 shm::genkey( std::string &key,
101  const std::size_t length )
102 {
103  if( length == 0 )
104  {
105  throw invalid_key_exception( "Key length must be longer than zero!" );
106  }
107  else if ( length > NAME_MAX )
108  {
109  std::stringstream errstream;
110  errstream << "Key length must be less than " << NAME_MAX << " characters, exiting!";
111  throw invalid_key_exception( errstream.str() );
112  }
113  using key_t = std::uint32_t;
114  FILE *fp = std::fopen("/dev/urandom","r");
115  if(fp == NULL)
116  {
117  const char *err = "Error, couldn't open /dev/urandom!!\n";
118  std::perror( err );
119  exit( EXIT_FAILURE );
120  }
121  /** figure out how many ints we'll need **/
122  const auto nints(
123  static_cast< std::size_t >(
124  static_cast< float >( length ) /
125  static_cast< float >( sizeof( std::uint64_t ) ) ) * 2 );
126  auto *intptr( new std::uint64_t[ nints ] );
127  if( std::fread( intptr, sizeof( std::uint64_t ), nints, fp) != nints )
128  {
129  /** BAIL OUT **/
130  const char *err = "Error, incorrect number of ints retured!!\n";
131  std::fclose(fp);
132  delete[]( intptr );
133  std::perror(err);
134  exit( EXIT_FAILURE );
135  }
136  else
137  {
138  std::fclose( fp );
139  }
140  std::size_t bytes_left( length );
141  std::uint64_t *intptr_64( intptr );
142  char * const buffer( (char*) malloc( ( sizeof( char ) * length ) + 1 ) );
143  std::memset( buffer, '\0', (sizeof( char ) * length) + 1 );
144  char *buff_ptr( buffer );
145  while( bytes_left > 8 )
146  {
147  std::snprintf( buff_ptr, 8, "%" PRIu64 "", (*intptr_64) );
148  ++intptr_64;
149  bytes_left -= 7;
150  buff_ptr += 7;
151  }
152  auto *intptr_32( reinterpret_cast< std::uint32_t* >( intptr_64 ) );
153  while( bytes_left > 4 )
154  {
155  std::snprintf( buff_ptr, 4, "%" PRIu32 "", (*intptr_32) );
156  bytes_left -= 3;
157  buff_ptr += 3;
158  ++intptr_32;
159  }
160  auto *intptr_16( reinterpret_cast< std::uint16_t* >( intptr_32 ) );
161  while( bytes_left >= 1 )
162  {
163  std::snprintf( buff_ptr, 2, "%" PRIu16 "", (*intptr_16) );
164  bytes_left -= 1;
165  buff_ptr += 1;
166  ++intptr_16;
167  }
168  delete[]( intptr );
169  key = std::string( buffer );
170  std::free( buffer );
171  return;
172 }

174 void*
175 shm::init( const std::string &key,
176  const std::size_t nbytes,
177  const bool zero /* zero mem */,
178  void *ptr )
179 {
180  if( nbytes == 0 )
181  {
182  throw bad_shm_alloc( "nbytes cannot be zero when allocating memory!" );
183  }
184  int fd( shm::failure );
185  errno = success;
186  /* essentially we want failure if the file exists already */
187  if( access( key.c_str(), F_OK ) == shm::success )
188  {
189  std::stringstream ss;
190  ss << "File exists with name \"" << key << "\", error code returned: ";
191  ss << std::strerror( errno );
192  throw bad_shm_alloc( ss.str() );
193  }
194  /* set read/write set create if not exists */
195  const std::int32_t flags( O_RDWR | O_CREAT | O_EXCL );
196  /* set read/write by user */
197  const mode_t mode( S_IWUSR | S_IRUSR );
198  errno = shm::success;
199  fd = shm_open( key.c_str(),
200  flags,
201  mode );
202  if( fd == failure )
203  {
204  std::stringstream ss;
205  ss << "Failed to open shm with file descriptor \"" <<
206  key << "\", error code returned: ";
207  ss << std::strerror( errno );
208  throw bad_shm_alloc( ss.str() );
209  }
210 
211  /** before truncation, lets do a sanity check **/
212  const auto num_phys_pages( sysconf( _SC_PHYS_PAGES ) );
213  const auto page_size( sysconf( _SC_PAGE_SIZE ) );
214  const auto total_possible_bytes( num_phys_pages * page_size );
215  if( nbytes > total_possible_bytes )
216  {
217  std::stringstream errstr;
218  errstr << "You've tried to allocate too many bytes (" << nbytes << "),"
219  << " the total possible is (" << total_possible_bytes << ")\n";
220  throw bad_shm_alloc( errstr.str() );
221  }
222  /* else begin truncate */
223  /* else begin mmap */
224  /** get allocations size including extra dummy page **/
225  const auto alloc_bytes(
226  static_cast< std::size_t >(
227  std::ceil(
228  static_cast< float >( nbytes) /
229  static_cast< float >( page_size ) ) + 1 ) * page_size
230  );
231  errno = shm::success;
232  if( ftruncate( fd, alloc_bytes ) != shm::success )
233  {
234  std::stringstream ss;
235  ss << "Failed to truncate shm for file descriptor (" << fd << ") ";
236  ss << "with number of bytes (" << nbytes << "). Error code returned: ";
237  ss << std::strerror( errno );
238  shm_unlink( key.c_str() );
239  throw bad_shm_alloc( ss.str() );
240  }
241  /**
242  * NOTE: actual allocation size should be alloc_bytes,
243  * user has no idea so we'll re-calc this at the end
244  * when we unmap the data.
245  */
246  errno = shm::success;
247  void *out( nullptr );
248  out = mmap( ptr,
249  alloc_bytes,
250  ( PROT_READ | PROT_WRITE ),
251  MAP_SHARED,
252  fd,
253  0 );
254  if( out == MAP_FAILED )
255  {
256  std::stringstream ss;
257  ss << "Failed to mmap shm region with the following error: " <<
258  std::strerror( errno ) << ",\n" << "unlinking.";
259  shm_unlink( key.c_str() );
260  throw bad_shm_alloc( ss.str() );
261  }
262  /** mmap should theoretically return start of page **/
263  assert( reinterpret_cast< std::uintptr_t >( out ) % page_size == 0 );
264  if( zero )
265  {
266  /* everything theoretically went well, lets initialize to zero */
267  std::memset( out, 0x0, nbytes );
268  }
269  char *temp( reinterpret_cast< char* >( out ) );
270  if( mprotect( (void*) &temp[ alloc_bytes - page_size ],
271  page_size,
272  PROT_NONE ) != 0 )
273  {
274 #if DEBUG
275  perror( "Error, failed to set page protection, not fatal just dangerous." );
276 #endif
277  }
278  return( out );
279 }

281 void*
282 shm::open( const std::string &key )
283 {
284  /* accept no zero length keys */
285  assert( key.length() > 0 );
286  int fd( shm::failure );
287  const int flags( O_RDWR | O_CREAT );
288  mode_t mode( 0 );
289  errno = success;
290  fd = shm_open( key.c_str(),
291  flags,
292  mode );
293  if( fd == failure )
294  {
295  std::stringstream ss;
296  ss << "Failed to open shm with key \"" << key << "\", with the following error code: ";
297  ss << std::strerror( errno );
298  throw bad_shm_alloc( ss.str() );
299  }
300  struct stat st;
301  std::memset( &st,
302  0x0,
303  sizeof( struct stat ) );
304  /* stat the file to get the size */
305  if( fstat( fd, &st ) != shm::success )
306  {
307  std::stringstream ss;
308  ss << "Failed to stat shm region with the following error: " << std::strerror( errno ) << ",\n";
309  ss << "unlinking.";
310  shm_unlink( key.c_str() );
311  throw bad_shm_alloc( ss.str() );
312  }
313  void *out( nullptr );
314  errno = shm::success;
315  out = mmap( nullptr,
316  st.st_size,
317  (PROT_READ | PROT_WRITE),
318  MAP_SHARED,
319  fd,
320  0 );
321  if( out == MAP_FAILED )
322  {
323  std::stringstream ss;
324  ss << "Failed to mmap shm region with the following error: " << std::strerror( errno ) << ",\n";
325  ss << "unlinking.";
326  shm_unlink( key.c_str() );
327  throw bad_shm_alloc( ss.str() );
328  }
329  /* close fd */
330  ::close( fd );
331  /* done, return mem */
332  return( out );
333 }

335 bool
336 shm::close( const std::string &key,
337  void **ptr,
338  const std::size_t nbytes,
339  const bool zero,
340  const bool unlink )
341 {
342  if( ptr != nullptr )
343  {
344  if( zero && ( *ptr != nullptr ) )
345  {
346  std::memset( *ptr, 0x0, nbytes );
347  }
348  /** get allocations size including extra dummy page **/
349  const auto page_size( sysconf( _SC_PAGESIZE ) );
350  const auto alloc_bytes(
351  static_cast< std::size_t >(
352  std::ceil(
353  static_cast< float >( nbytes ) /
354  static_cast< float >( page_size ) ) + 1 ) * page_size
355  );
356  errno = shm::success;
357  if( ( *ptr != nullptr ) && ( munmap( *ptr, alloc_bytes ) != shm::success ) )
358  {
359 #if DEBUG
360  perror( "Failed to unmap shared memory, attempting to close!!" );
361 #endif
362  }
363  *ptr = nullptr;
364  }
365  if( unlink )
366  {
367  if( shm_unlink( key.c_str() ) != shm::success )
368  {
369  switch( errno )
370  {
371  case( ENOENT ):
372  {
373  throw invalid_key_exception( "File descriptor to unlink does not exist!" );
374  }
375  default:
376  throw invalid_key_exception( "Undefined error, check error codes" );
377  }
378  }
379  }
380  return( true );
381 }

383 bool
384 shm::move_to_tid_numa( const pid_t thread_id,
385  void *ptr,
386  const std::size_t nbytes )
387 {
388 #if __linux && ( PLATFORM_HAS_NUMA == 1 )
389  /** check alignment of pages first **/
390  const auto page_size( sysconf( _SC_PAGESIZE ) );
391 
392  const auto ptr_addr(
393  reinterpret_cast< std::uintptr_t >( ptr ) );
394  if( (ptr_addr % page_size) != 0 )
395  {
396  std::stringstream ss;
397  ss << "Variable 'ptr' must be page aligned, currently it is(" <<
398  ptr_addr % page_size << "), off please fix...exiting!!\n";
399  throw page_alignment_exception( ss.str() );
400  }
401 
402  /** check to see if NUMA avail **/
403  if( numa_available() == -1 )
404  {
405  return( true );
406  }
407  /** first check to see if there is more than one numa node **/
408  const auto num_numa( numa_num_configured_nodes() );
409  if( num_numa == 1 )
410  {
411  /** no point in continuing **/
412  return( true );
413  }
414 
415  /** RETURN VAL HERE */
416  bool moved( false );
417  /** get the numa node of the calling thread **/
418  const auto local_node( numa_node_of_cpu( sched_getcpu() ) );
419  /** get numa node that 'pages' is on **/
420  /**
421  * Note: only way I could figure out how to do this was get a cpu
422  * set, we'll check to see how many CPU's are in the CPU set of
423  * the thread, then we'll see if all those CPU's are on the
424  * same NUMA node, if they are then we'll check to see if the
425  * alloc is on the same node, otherwise we'll move the pages to that
426  * node. The interesting case comes when there are multiple
427  * CPUs per set and multiple possible NUMA nodes to choose from. The simple
428  * answer, the one I'm taking a the moment is that we can't
429  * really decide in this function. I think in the future it'd
430  * be good to have a profiler decide the shortest latency and
431  * highest bandwidth for long running programs at startup and
432  * do some sort of graph matching.
433  */
434  cpu_set_t *cpuset( nullptr );
435  /** we want set to include all cpus **/
436  const auto num_cpus_alloc( get_nprocs_conf() );
437  std::size_t cpu_allocate_size( -1 );
438 #if (__GLIBC_MINOR__ > 9 ) && (__GLIBC__ == 2)
439  cpuset = CPU_ALLOC( num_cpus_alloc );
440  assert( cpuset != nullptr );
441  cpu_allocate_size = CPU_ALLOC_SIZE( num_cpus_alloc );
442  CPU_ZERO_S( cpu_allocate_size, cpuset );
443 #else
444  cpu_allocate_size = sizeof( cpu_set_t );
445  cpuset = reinterpret_cast< cpu_set_t* >( malloc( cpu_allocate_size ) );
446  assert( cpuset != nullptr );
447  CPU_ZERO( cpuset );
448 #endif
449  /** now get affinity **/
450  if( sched_getaffinity( thread_id, cpu_allocate_size, cpuset ) != 0 )
451  {
452  perror( "Failed to get affinity for calling thread!" );
453  return( false );
454  }
455  /**
456  * NOTE: for the moment we're using an extremely simplistic
457  * distance metric, we should probably be using the numa_distance
458  * combined with the actual latancy/bandwidth between the
459  * processor cores.
460  */
461  auto numa_node_count( 0 );
462  auto denom( 0 );
463  for( auto p_index( 0 ); p_index < num_cpus_alloc; p_index++ )
464  {
465  if( CPU_ISSET( p_index /* cpu */, cpuset ) )
466  {
467  const auto ret_val( numa_node_of_cpu( p_index ) );
468  if( ret_val != -1 )
469  {
470  numa_node_count += ret_val;
471  denom++;
472  }
473  }
474  }
475  const auto target_node( numa_node_count / denom );
476 
477  /** we need to know how many pages we have **/
478  const auto num_pages(
479  static_cast< std::size_t >(
480  std::ceil(
481  static_cast< float >( nbytes ) / static_cast< float >( page_size ) ) ) );
482  using int_t = std::int32_t;
483  int_t *mem_node ( new int_t[ num_pages ] );
484  int_t *mem_status( new int_t[ num_pages ] );
485  void **page_ptr = (void**)malloc( sizeof( void* ) * num_pages );
486  char *temp_ptr( reinterpret_cast< char* >( ptr ) );
487  for( auto node_index( 0 ),
488  page_index( 0 ) /** init **/;
489  node_index < num_pages /** cond **/;
490  node_index++, page_index += page_size /** incr **/)
491  {
492  mem_node [ node_index ] = target_node;
493  mem_status[ node_index ] = -1;
494  page_ptr[ node_index ] = (void*)&temp_ptr[ page_index ];
495  }
496  if( move_pages( thread_id /** thread we want to move for **/,
497  num_pages /** total number of pages **/,
498  page_ptr /** pointers to the start of each page **/,
499  mem_node /** node we want to move to **/,
500  mem_status/** status flags, check for debug **/,
501  MPOL_MF_MOVE ) != 0 )
502  {
503 #if DEBUG
504  perror( "failed to move pages, non-fatal error but results may vary.");
505  /** TODO, re-code all status flag checks..accidentally over wrote **/
506 #endif
507  moved = false;
508  }
509  delete[]( mem_node );
510  delete[]( mem_status );
511  free( page_ptr );
512  return( moved );
513 #else /** no NUMA avail **/
514  return( false );
515 #endif
516 }
