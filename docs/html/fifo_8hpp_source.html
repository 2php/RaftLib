89  /**
90  * allocate - returns a reference to a writeable
91  * member at the tail of the FIFO. You must have
92  * a subsequent call to send in order to release
93  * this object to the FIFO once it is written.
94  * If the user needs to de-allocate the memory without
95  * using it, they can call the deallocate function.
96  * @return T&
97  */
98  template < class T,
99  typename std::enable_if<
100  inline_nonclass_alloc< T >::value >::type* = nullptr >
101  T& allocate()
102  {
103  void *ptr( nullptr );
104  /** call blocks till an element is available **/
105  local_allocate( &ptr );
106  return( *( reinterpret_cast< T* >( ptr ) ) );
107  }

111  class ... Args,

113  inline_class_alloc< T >::value >::type* = nullptr >

116  {
117  void *ptr( nullptr );

119  local_allocate( &ptr );

196  /**
197  * allocate_range - returns a std::vector of references to
198  * n items on the queue. If for some reason the capacity
199  * of the queue is less than n or some other unspecified
200  * error occurs then the number allocated is returned in
201  * n_ret. To release items to the queue, use push_range
202  * as opposed to the standard push.
203  * @param n - const std::size_t, # items to allocate
204  * @return std::vector< std::reference_wrapper< T > >
205  */
206  template < class T >
207  auto allocate_range( const std::size_t n ) ->
208  std::vector< std::reference_wrapper< T > >
209  {
210  std::vector< std::reference_wrapper< T > > output;
211  void *ptr( (void*) &output );
212  local_allocate_n( ptr, n );
213  /** compiler should optimize this copy, if not then it'll be a copy of referneces not full objects **/
214  return( std::forward<
215  std::vector<
216  std::reference_wrapper< T > > >( output ) );
217  }
