75 void
76 raft::map::enableDuplication( kernelkeeper &source, kernelkeeper &all )
77 {
78  auto &source_k( source.acquire() );
79  auto &all_k ( all.acquire() );
80  /** don't have to do this but it makes it far more apparent where it comes from **/
81  void * const kernel_ptr( reinterpret_cast< void* >( &all_k ) );
82  using kernel_ptr_t = std::remove_reference< decltype( all_k ) >::type;
83  /** need to grab impl of Lengauer and Tarjan dominators, use for SESE **/
84  /** in the interim, restrict to kernels that are simple to duplicate **/
85  /** NOTE: there's a better linear SESE algorithm jcb17May16 **/
86  GraphTools::BFS( source_k,
87  []( PortInfo &a, PortInfo &b, void *data )
88  {
89  auto * const all_k( reinterpret_cast< kernel_ptr_t* >( data ) );
90  if( a.out_of_order && b.out_of_order )
91  {
92  /** case of inline kernel **/
93  if( b.my_kernel->input.count() == 1 &&
94  b.my_kernel->output.count() == 1 &&
95  a.my_kernel->dup_candidate )
96  {
97  auto *kernel_a( a.my_kernel );
98  assert( kernel_a->input.count() == 1 );
99  auto &port_info_front( kernel_a->input.getPortInfo() );
100  auto *front( port_info_front.other_kernel );
101  auto &front_port_info( front->output.getPortInfo() );
102  /**
103  * front -> kernel_a goes to
104  * front -> split -> kernel_a
105  */
106  auto *split(
107  static_cast< raft::kernel* >(
108  port_info_front.split_func() ) );
109  all_k->insert( split );
110  MapBase::insert( front, front_port_info,
111  kernel_a, port_info_front,
112  split );
113 
114  assert( kernel_a->output.count() == 1 );
115 
116  /**
117  * now we need the port info from the input
118  * port on back
119  **/
120 
121  /**
122  * kernel_a -> back goes to
123  * kernel_a -> join -> back
124  */
125  auto *join( static_cast< raft::kernel* >( a.join_func() ) );
126  all_k->insert( join );
127  MapBase::insert( a.my_kernel, a,
128  b.my_kernel, b,
129  join );
130  /**
131  * finally set the flag to the scheduler
132  * so that the parallel map manager can
133  * pick it up an use it.
134  */
135  a.my_kernel->dup_enabled = true;
136  }
137  /** parallalizable source, single output no inputs**/
138  else if( a.my_kernel->input.count() == 0 &&
139  a.my_kernel->output.count() == 1 )
140  {
141  auto *join( static_cast< raft::kernel* >( a.join_func() ) );
142  all_k->insert( join );
143  MapBase::insert( a.my_kernel, a,
144  b.my_kernel, b,
145  join );
146  a.my_kernel->dup_enabled = true;
147  }
148  /** parallelizable sink, single input, no outputs **/
149  else if( b.my_kernel->input.count() == 1 &&
150  b.my_kernel->output.count() == 0 )
151  {
152  auto *split(
153  static_cast< raft::kernel* >( b.split_func() ) );
154  all_k->insert( split );
155  MapBase::insert( a.my_kernel, a,
156  b.my_kernel, b,
157  split );
158  b.my_kernel->dup_enabled = true;
159  }
160  /**
161  * flag as candidate if the connecting
162  * kernel only has one input port.
163  */
164  else if( b.my_kernel->input.count() == 1 )
165  {
166  /** simply flag as a candidate **/
167  b.my_kernel->dup_candidate = true;
168  }
169 
170  }
171  },
172  kernel_ptr,
173  false );
174  source.release();
175  all.release();
176 }

247 kernel_pair_t
248 raft::map::operator += ( kpair &p )
249 {
250  kpair * const pair( &p );
251  assert( pair != nullptr );
252 
253  /** start at the head, go forward **/
254  kpair *next = pair->head;
255  assert( next != nullptr );
256 
257  /** init return struct **/
258  kernel_pair_t ret_kernel_pair;
259 
260  /**
261  * set to true when we've done the first pass
262  * so that the ret_kernel_pair has the correct
263  * source kernels for this addition to the map
264  * most of the time this will look just like
265  * a '>>' op, but others this will get quite
266  * complicated
267  */
268  bool initialized( false );
269 
270  /** used to do nested splits **/
271  enum sj_t : std::int8_t { split, join, cont };
272  const static std::array< std::string, 3 > sj_t_str(
273  {{ "split", "join", "cont" }}
274  );
275 
276  auto next_link_type( []( const kpair * const k ) -> sj_t
277  {
278  if( k->split_to && ! k->join_from )
279  {
280  return( split );
281  }
282  else if( ! k->split_to && k->join_from )
283  {
284  return( join );
285  }
286  else if( ! k->split_to && ! k->join_from )
287  {
288  return( cont );
289  }
290  else
291  {
292  throw InvalidTopologyOperationException( "Invalid ordering of topology operators \"<=\", \">>\", \">=.\" Please check and try again\n" );
293  }
294  //should never reach here **/
295  assert( false );
296  return( cont );
297  });
298 
299  std::stack< sj_t > stack;
300  /**
301  * dup kernels arranged in "groups" so that
302  * a =< b
303  * a -> b1
304  * a -> b2
305  * and b^n is one group
306  * each kernel attached to b^n is
307  * then chained group to group
308  * until we reduce
309  */
310  kernels_t groups;
311 
312  /**
313  * keep # of out-edges for splits
314  */
315  split_stack_t split_out_edges;
316 
317  /** could throw an exception so use push_back **/
318  groups.push_back( up_group_t( new group_t( ) ) );
319  /**
320  * start kernels vector off
321  * emplace used since kernel can't throw an exception
322  */
323  groups.back()->emplace_back( next->src );
324 
325  /** hold on to these for a bit so we can back track if needed **/
326  std::vector< kpair* > kpair_store;
327 
328  /** loop over chain **/
329  while( next != nullptr )
330  {
331  stack.push( next_link_type( next ) );
332  decltype( groups ) temp_groups;
333  switch( stack.top() )
334  {
335  case( cont ):
336  {
337  inline_cont( groups,
338  temp_groups,
339  next );
340  //done, move on
341  stack.pop();
342  }
343  break;
344  case( split ):
345  {
346  inline_split( split_out_edges,
347  groups,
348  temp_groups,
349  next );
350  }
351  break;
352  case( join ):
353  {
354  //make sure something bad didn't happen
355  stack.pop();
356  if( stack.size() == 0 /** mult kernels going ot single kernel **/ )
357  {
358  /**
359  * whoops, looks like we need to clear the source
360  * ret_kernel_pair. also reset initialized so we
361  * capture all the new source kernels that are added.
362  */
363  ret_kernel_pair.clearSrc();
364  initialized = false;
365  inline_dup_join( groups,
366  temp_groups,
367  next );
368  }
369  else if( stack.top() == split )
370  {
371  //can pop now, don't need the old split
372  stack.pop();
373  inline_join( split_out_edges,
374  groups,
375  temp_groups,
376  next );
377  }
378  else
379  {
380  assert( false );
381  }
382  }
383  break;
384  default:
385  assert( false );
386  }
387 
388  if( ! initialized )
389  {
390  for( /** unique ptr **/ auto &groups_ptr : groups )
391  {
392  for( auto * const kernel : (*groups_ptr) )
393  {
394  assert( kernel != nullptr );
395  ret_kernel_pair.addSrc( *kernel );
396  }
397  }
398  initialized = true;
399  }
400  /** don't need these anymore **/
401  groups.clear();
402  /** now old dst (in temp_groups) become new sources **/
403  groups = std::move( temp_groups );
404 
405  kpair * const temp_kpair = next;
406  next = next->next;
407  kpair_store.emplace_back( temp_kpair );
408  }
409 
410  /**
411  * groups should now have the final destinations across
412  * multiple groups potentially. Take these, flatten to
413  * a single destination for the return vector since this
414  * is what the specification expects (i.e., all actual
415  * end destinations added to map....which are the last
416  * kernels added or leaf nodes).
417  */
418  for( /** unique ptr **/ auto &groups_ptr : groups )
419  {
420  for( auto * const kernel : (*groups_ptr) )
421  {
422  assert( kernel != nullptr );
423  ret_kernel_pair.addDst( *kernel );
424  }
425  }
426 
427 
428 
429  for( auto *ptr : kpair_store )
430  {
431  delete( ptr );
432  }
433 
434  /**
435  * TODO, need to get the kernel_pair_t to
436  * return the full list of kernels. Basically
437  * need to go back and pass the kernel_pair_t
438  * through the split/cont/join in order ot
439  * get the right kernels into the list, especially
440  * ones that are cloned internally otherwise
441  * the user might not get the returns that
442  * the expected.
443  */
444  return( ret_kernel_pair );
445 }

447 void
448 raft::map::inline_cont( kernels_t &groups,
449  kernels_t &temp_groups,
450  kpair * const next )
451 {
452  /**
453  * for each kernel, and each group make an output
454  * link inline
455  */
456  for( auto &group : groups )
457  {
458  /** we need to insert a new group in temp_groups **/
459  temp_groups.push_back( up_group_t( new group_t() ) );
460  for( auto *kernel : *group )
461  {
462  assert( kernel != nullptr );
463  if( temp_groups[ 0 ]->size() == 0 )
464  {
465  //simply join source do destination
466  joink( next );
467  //emplace back temp_groups kernel
468  temp_groups[ 0 ]->emplace_back( next->dst );
469  }
470  else
471  {
472  auto * const dst_clone( next->dst->clone() );
473  next->src = kernel;
474  next->dst = dst_clone;
475  joink( next );
476  temp_groups.back()->emplace_back( dst_clone );
477  }
478  }
479  }
480 
481  return;
482 }

484 void
485 raft::map::inline_split( split_stack_t &split_stack,
486  kernels_t &groups,
487  kernels_t &temp_groups,
488  kpair * const next )
489 {
490  split_stack.push( next->src_out_count );
491  /**
492  * for each kernel, create a destination
493  * for each of it's output ports, link it
494  */
495  for( auto &group : groups )
496  {
497  /** each group will have the same # of out-edges, save **/
498  for( auto *kernel : *group )
499  {
500  //TODO: throw an exception for
501  //only one port, user meant
502  //to use raft::out likely not
503  //the static split
504 
505  //TODO: throw exception for more
506  //than one input port for destination
507  //won't work with split/join
508  temp_groups.push_back( up_group_t( new group_t() ) );
509  //for each output port in kernel
510  if( temp_groups[ 0 ]->size() == 0 )
511  {
512  auto &port( kernel->output );
513  for( auto it( port.begin() );
514  it != port.end(); ++it )
515  {
516  next->src = kernel;
517  //next->dst already eq dst
518  next->src_name = it.name();
519  next->has_src_name = true;
520  if( it == port.begin() )
521  {
522  temp_groups.back()->emplace_back( next->dst );
523  }
524  else
525  {
526  auto * const dst_clone( next->dst->clone() );
527  next->dst = dst_clone;
528  //dst name should be same as first
529  temp_groups.back()->emplace_back( dst_clone );
530  }
531  joink( next );
532  }
533  }
534  else
535  {
536  auto &port( kernel->output );
537  for( auto it( port.begin() );
538  it != port.end(); ++it )
539  {
540  next->src = kernel;
541  //next->dst already eq dst
542  next->src_name = it.name();
543  next->has_src_name = true;
544  //now we always need to clone
545  auto * const dst_clone( next->dst->clone() );
546  next->dst = dst_clone;
547  //dst name should be same as first
548  temp_groups.back()->emplace_back( dst_clone );
549  joink( next );
550  }
551  }
552  }
553  }
554  return;
555 }

557 void
558 raft::map::inline_join( split_stack_t &split_stack,
559  kernels_t &groups,
560  kernels_t &temp_groups,
561  kpair * const next )
562 {
563  /** need to get the count **/
564  std::size_t count( 0 );
565  /**
566  * for every kernel in each group, join that
567  * group's kernels to a single destination and
568  * "join"
569  */
570  auto joinfunc( [&]( up_group_t &group, kpair *n )
571  {
572  auto it( group->begin() );
573  for( auto port_it( n->dst->input.begin() );
574  port_it != n->dst->input.end(); ++port_it )
575  {
576  n->src = (*it);
577  n->dst_name = port_it.name();
578  n->has_dst_name = true;
579  joink( n );
580  ++it;
581  }
582 
583  });
584 
585  for( auto &group : groups )
586  {
587  /**
588  * need to add exception for too many
589  * output ports for source, or too
590  * few input ports for destination
591  */
592 
593  if( count <= 0 )
594  {
595  temp_groups.push_back( up_group_t( new group_t() ) );
596  count = split_stack.top();
597  }
598  if( temp_groups[ 0 ]->size() == 0 )
599  {
600  /** use first kernel **/
601  temp_groups.back()->emplace_back( next->dst );
602  /**
603  * take all kernels from first group, attach
604  * to current dst. kernel
605  */
606  joinfunc( group, next );
607  }
608  else
609  {
610  /**
611  * we need to clone a kernel
612  */
613  auto *temp_groups_k( next->dst->clone() );
614  next->dst = temp_groups_k;
615  temp_groups.back()->emplace_back( temp_groups_k );
616  joinfunc( group, next );
617  }
618  count--;
619  }
620  split_stack.pop();
621 }

624 void
625 raft::map::inline_dup_join( kernels_t &groups,
626  kernels_t &temp_groups,
627  kpair * const next )
628 {
629  /**
630  * only one group at the moment, this pre-condition
631  * could change if we make the duplication code
632  * a bit more fancy. need to abstract away more
633  * to scale better. - jcb
634  */
635  assert( groups.size() == 1 );
636  temp_groups.push_back( up_group_t( new group_t() ) );
637  up_group_t &gp( groups[ 0 ] );
638 
639  /** should currently have only one kernel here **/
640  assert( gp->size() == 1 );
641  /** get count to satisfy postcondition at end **/
642  const auto dst_inport_count( next->dst->input.count() );
643 
644  /**
645  * this is a bit confusing so here's the overall scheme:
646  * - for each input port in next's dst, create a souce.
647  * there will of course already be one source but we'll
648  * likely need to add more. Each of these new kernels
649  * will need to have the name set, and the dst. will need
650  * to have the name set so that we don't throw any
651  * exceptions. -jcb
652  */
653  for( auto port_it( next->dst->input.begin() );
654  port_it != next->dst->input.end(); ++port_it )
655  {
656  if( port_it == next->dst->input.begin() )
657  {
658  /** connect source to dest **/
659  next->dst_name = port_it.name();
660  next->has_dst_name = true;
661  joink( next );
662  temp_groups.back()->emplace_back( next->dst );
663  /**
664  * at this point a single chain has been added
665  * but we need to check for the next one to make
666  * sure that we don't need to duplicate from head
667  */
668  }
669  else
670  {
671  /**
672  * need to find out if there's more than one kernel in front
673  * of this one. don't need to keep track of the new kernels
674  * here.
675  */
676  raft::kernel *src( nullptr );
677  if( next->head != next )
678  {
679  /** go back to head **/
680  auto *head_next( next->head );
681  /** chain condition, start from the source of this kpair **/
682  while( head_next != next )
683  {
684  if( src == nullptr )
685  {
686  /** clone the sosurce **/
687  src = head_next->src->clone();
688  gp->emplace_back( src );
689  }
690  auto *dst( head_next->dst->clone() );
691  head_next->src = src;
692  head_next->dst = dst;
693  joink( head_next );
694  src = dst;
695  head_next = next;
696  }
697  /** now need to link chain to current join **/
698  //src already alloced
699  assert( src != nullptr );
700  next->dst_name = port_it.name();
701  next->has_dst_name = true;
702  next->src = src;
703  joink( next );
704  }
705  else /** only one kernel **/
706  {
707  //need to duplicate source
708  next->src = next->src->clone();
709  gp->emplace_back( next->src );
710  next->dst_name = port_it.name();
711  next->has_dst_name = true;
712  joink( next );
713  }
714  }
715  }
716  /**
717  * post condition, gp should have # of kernels that match dest in-port
718  * count.
719  */
720  assert( gp->size() == dst_inport_count );
721  return;
722 }
